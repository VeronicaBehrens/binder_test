Exercise 2: Mean expression of a gene across phenotypes¶

In [3]:
# Pick the first gene in the expression matrix
gene_name <- rownames(expr_data)[1]
cat("Gene:", gene_name, "\n\n")

# Get expression values for this gene across all cells
gene_expr <- as.numeric(expr_data[1, ])

# Match cell IDs between expr_data columns and cell_meta
cell_ids <- colnames(expr_data)
meta_idx <- match(cell_ids, cell_meta$cell_id)
phenotypes <- cell_meta$phenotype[meta_idx]

# Calculate mean expression per phenotype using tapply
mean_expr_by_pheno <- tapply(gene_expr, phenotypes, mean)

cat("Mean expression of", gene_name, "by phenotype:\n")
print(round(sort(mean_expr_by_pheno, decreasing = TRUE), 1))

Exercise 3: Boxplot of total_features by phenotype¶

In [4]:
# Create a color palette — one color per phenotype
n_pheno <- length(unique(cell_meta$phenotype))
colors <- rainbow(n_pheno, s = 0.6, v = 0.8)

# Boxplot
boxplot(total_features ~ phenotype,
        data = cell_meta,
        las = 2,
        col = colors,
        main = "Total Features per Cell by Differentiation Stage",
        xlab = "",
        ylab = "Total Features (detected genes)")

# Add horizontal line at overall median
abline(h = median(cell_meta$total_features), col = "darkred", lwd = 2, lty = 2)

# Add a legend for the median line
legend("topright", legend = "Overall median", lty = 2, col = "darkred", lwd = 2)

Exercise 4: Top 10 expressed genes to CSV¶

In [5]:
# Compute mean expression for each gene (row means)
gene_means <- rowMeans(expr_data)

# Sort in descending order
sorted_genes <- sort(gene_means, decreasing = TRUE)

# Get top 10
top10 <- sorted_genes[1:10]

# Create a data frame
top10_df <- data.frame(
  gene_id = names(top10),
  mean_expression = round(as.numeric(top10), 2),
  stringsAsFactors = FALSE
)

print(top10_df)

# Write to CSV
write.csv(top10_df, "top10_expressed_genes.csv", row.names = FALSE)
cat("\nSaved to top10_expressed_genes.csv\n")
